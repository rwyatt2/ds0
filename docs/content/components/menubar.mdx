---
title: Menubar
description: A horizontal menu bar for desktop application navigation.
---
## Usage
```tsx
import { Menubar } from '@ds0/react/menubar';
<Menubar>
  <Menubar.Menu>
    <Menubar.Trigger>File</Menubar.Trigger>
    <Menubar.Content>
      <Menubar.Item>New</Menubar.Item>
      <Menubar.Item>Open</Menubar.Item>
    </Menubar.Content>
  </Menubar.Menu>
</Menubar>
```
